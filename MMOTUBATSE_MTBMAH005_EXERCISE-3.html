cities = {
    "Walvis Bay": (14.5, -23),
    "Cape Town": (18.4, -34),
    "Rio de Janeiro": (-43.2, -23),
    "Montevideo": (-56.2, -35)
}

fig = plt.figure(figsize=(12, 8))
ax = plt.axes(projection=ccrs.Mercator())

ax.set_extent([-70, 30, -50, -20], crs=ccrs.PlateCarree())

ax.add_feature(cfeature.LAND, facecolor='lightgray')
ax.add_feature(cfeature.OCEAN, facecolor='aliceblue')
ax.add_feature(cfeature.COASTLINE, linewidth=0.8)
cfeature.NaturalEarthFeature('physical', 'land', '110m')

gl = ax.gridlines(draw_labels=True,
                  linewidth=0.6,
                  color='gray',
                  alpha=0.6,
                  linestyle='--')

gl.top_labels = False
gl.right_labels = False
gl.xlabel_style = {'size': 10}
gl.ylabel_style = {'size': 10}

for city, (lon, lat) in cities.items():
    ax.plot(lon, lat,
            marker='*',
            color='black',
            markersize=5,
            transform=ccrs.PlateCarree())

    ax.text(lon + 2, lat,
            city,
            fontsize=9,
            transform=ccrs.PlateCarree())

    ax.text(0.94, 0.78, 'SOUTHERN AFRICA',
        transform=ax.transAxes,
        fontsize=10,
        ha='center',
        va='center')

    ax.text(0.10, 0.80, 'SOUTHERN SOUTH AMERICA',
        transform=ax.transAxes,
        fontsize=10,
        ha='center',
        va='center')

ax.text(0.50, 0.50, 'SOUTH ATLANTIC OCEAN',
